## dir_api

dir相关api主要用于处理文件目录功能，接口如下。

- opendir: <dirent.h>
- readdir: <dirent.h>
- closedir: <dirent.h>
- mkdir: <sys/stat.h>

详细信息如下。

```c
//基于路径获取目录指针
DIR *opendir (const char *__name);
    __name: 获取目录指针的路径
    成功返回获取的目录指针，否则为空

//获取目录下子项的指针
struct dirent *readdir (DIR *__dirp);
    __dirp: 已经请求的目录指针
    成功返回获取的目录内子项信息(包含目录和文件)，否则为空

//关闭获取的目录
int closedir (DIR *__dirp);
    __dirp: 已经请求的目录指针

//根据权限和路径创建目录
int mkdir (const char *__path, __mode_t __mode);
    __path: 创建目录的路径
    __mode: 创建目录的权限
    成功返回0，失败返回-1，并设置错误值
```

## pthread

```c
//线程的创建  
int pthread_create(pthread_t *tid, const pthread_arrt_t* attr,  
                    void*(*start_routine)(void *), void* arg);  
//等待线程的结束,非分离的线程在结束后只有执行join才会释放全部资源  
int pthread_join(pthread_t thread, void **retval);
//线程离开时的返回值(必须为malloc或者全局变量)  
void pthread_exit(void * retval);
//分离线程,被分离的线程在结束后自动释放所有资源  
int pthread_detach(pthread_t tid);
//多线程的交互--互斥量和自旋锁  
//互斥量 
pthread_mutex_t m_lock;  
pthread_mutex_destroy(&m_lock);  
thread_mutex_lock(&m_lock)  
pthread_mutex_unlock(&m_lock);  
//自旋锁  
pthread_spinlock_t m_spinlock;  
pthread_spin_destroy(&m_spinlock);  
pthread_spin_lock(&m_spinlock);  
pthread_spin_trylock(&m_spinlock);  
pthread_spin_unlock(&m_spinlock);  
```
